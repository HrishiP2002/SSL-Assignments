\section{Pseudo Code}

\begin{algorithmic}
\Function{QUICKSORT}{$A[ ],p,r$}
\If{$p<r$} 
   \State $q\gets \text{PARTITION}(A,p,r)$ 
   \State $\text{QUICKSORT}(A,p,q-1)$ 
   \State $\text{QUICKSORT}(A,q+1,r)$ 
\EndIf
\EndFunction



\Function{PARTITION}$(A[ ],p,r) x\longleftarrow A[p] i \longleftarrow p-1 $
\For{$j \gets {p to r-1}$}
 \If{$A[j]<x$}
	\State i++
	\State swap$(A[i],A[j])$
	\EndIf
\EndFor
\State swap$(A([i+1],A[r])$
\State \Return ($i+1$)
\EndFunction
\end{algorithmic}
The Algorithm is derived taking hint from \cite{hoare1962quicksort} .
